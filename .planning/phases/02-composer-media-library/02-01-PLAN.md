    ```typescript
    async function getPost({ id, clientId }: { id: string; clientId: string }) {
      const postsAndMedia = await db.query.posts.findFirst({
        where: and(eq(posts.id, id), eq(posts.clientId, clientId)),
        with: {
          media: true, // includes media with postId set
        },

    async function updatePost({
      id,
      clientId,
      text,
      title,
      mediaIds = [],
    }: {
      id: string;
      clientId: string;
      text?: string;
      title?: string;
      mediaIds?: string[];
    }) {
      // Update post fields
      const values: any = {};
      if (text !== undefined) values.text = text;
      if (title !== undefined) values.title = title;
      if (Object.keys(values).length === 0) return null;

      const result = await db.update(posts)
        .set({ ...values, updatedAt: new Date() })
        .where(and(eq(posts.id, id), eq(posts.clientId, clientId)))
        .returning();

      // Reattach media: first detach all (nullify postId) then attach new ones
      if (result.length) {
        if (mediaIds.length > 0) {
          // Remove previous postId for this post
          await db.update(media).set({ postId: null }).where(eq(media.postId, id));
          // Attach new media
          await db.update(media)
            .set({ postId: id })
            .where(inArray(media.id, mediaIds));
        } else {
          // Detach all media
          await db.update(media).set({ postId: null }).where(eq(media.postId, id));
        }
      }
      return result[0];
    }

        where: eq(posts.clientId, clientId),
        orderBy: (posts, { desc }) => [desc(posts.createdAt)],
        with: {
          media: true, // includes attached media
        },
      });
    }
    ```

    ```typescript
    // lib/r2.ts
    import { S3Client, PutObjectCommand, GetObjectCommand } from "@aws-sdk/client-s3";
    import { S3RequestPresigner } from "@aws-sdk/s3-request-presigner";
    import { randomUUID } from "node:crypto";

    const client = new S3Client({
      region: "auto",
      endpoint: process.env.R2_ENDPOINT, // e.g., https://<account>.r2.cloudflared.com
      credentials: {
        accessKeyId: process.env.R2_ACCESS_KEY_ID!,
        secretAccessKey: process.env.R2_SECRET_ACCESS_KEY!,
      },
      // TLS — verify = true; if using self-signed dev cert, enable false only locally
    });

    const presigner = new S3RequestPresigner({ ...client });

    async function generateUploadUrl({
      clientId,
      contentType,
      fileName,
      userId,
    }: {
      clientId: string;
      contentType: string;
      fileName: string;
      userId: string;
    }) {
      // Normalize filename for client isolation: clientId/uuid.ext preserves isolation.
      // Key structure: media/{clientId}/{uuid}.{ext}
      const ext = fileName.split('.').pop() || '';
      const key = `media/${clientId}/${randomUUID()}.${ext}`;

      const command = new PutObjectCommand({
        Bucket: process.env.R2_BUCKET_NAME!,
        Key: key,
        ContentType: contentType,
        // Optional: ACL "public-read" for direct serving
        ACL: 'public-read',
        // Lifecycle: configure expiration via R2 bucket rules (e.g., 90 days)
        Metadata: {
          client_id: clientId,
          user_id: userId,
        },
      });

      const presignedUrl = await presigner.presign(command, { expiresIn: 3600 }); // 1h
      return {
        presignedUrl,
        key,
        publicUrl: `https://<YOUR_R2_CNAME>/${key}`, // Client configures R2 cname
        clientId,
        contentType,
      };
    }

    export { generateUploadUrl };
    ```

    ```typescript
    export async function POST(req: NextRequest) {
      // Auth: better-auth session check
      const session = await auth.api.getSession({ headers: await headers() });
      if (!session) {
        return NextResponse.json({ error: "Unauthorized" }, { status: 401 });
      }
      // Client scoping: require active client
      const { activeClientId, userId } = await requireClientScope();

      const { contentType, fileName } = await req.json();
      if (!contentType || !fileName) {
        return NextResponse.json({ error: "Missing content-type or filename" }, { status: 400 });
      }

      // Media type validation: images only for Phase 2 (future: videos in Phase 4)
      const allowed = [
        "image/jpeg", "image/png", "image/webp", "image/gif",
        // video later: "video/mp4", "video/quicktime"
      ];
      if (!allowed.includes(contentType)) {
        return NextResponse.json({ error: `Unsupported media type ${contentType}` }, { status: 400 });

        fileName,
        userId,
      });

        publicUrl,
    ```

    ```typescript
    import { requireClientScope } from "@/lib/clients";
    import { getPost, updatePost } from "@/lib/posts";
    import { auth } from "@/lib/auth";
    import { headers } from "next/headers";

    export async function GET(req: NextRequest, { params }: { params: Promise<{ id: string }> }) {
      const session = await auth.api.getSession({ headers: await headers() });
      if (!session) {
        return NextResponse.json({ error: "Unauthorized" }, { status: 401 });
      }
      const { activeClientId } = await requireClientScope();
      const { id } = await params;
      const post = await getPost({ id, clientId: activeClientId });

      if (!post) {
        return NextResponse.json({ error: "Post not found or access denied" }, { status: 404 });
      }

      return NextResponse.json(post, { status: 200 });
    }

    export async function PATCH(req: NextRequest, { params }: { params: Promise<{ id: string }> }) {
      const session = await auth.api.getSession({ headers: await headers() });
      if (!session) {
        return NextResponse.json({ error: "Unauthorized" }, { status: 401 });
      }
      const { activeClientId } = await requireClientScope();
      const { id } = await params;
      const { text, title, mediaIds } = await req.json();

      const updated = await updatePost({ id, clientId: activeClientId, text, title, mediaIds });
      if (!updated) {
        return NextResponse.json({ error: "Post not found or access denied" }, { status: 404 });
      }

      return NextResponse.json(updated, { status: 200 });
    }
    ```
